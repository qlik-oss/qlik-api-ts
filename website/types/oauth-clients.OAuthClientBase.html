    Type Alias OAuthClientBase

    Base schema for OAuth client requests and responses

    type OAuthClientBase = {
        allowedGrantTypes?: (
            "client_credentials"
            | "urn:qlik:oauth:user-impersonation"
        )[];
        allowedScopes?: string[];
        appType: "web" | "native" | "spa" | "anonymous-embed";
        clientName: string;
        clientUri?: string;
        createdAt: string;
        deletedAt?: string;
        description?: string;
        disableTag?: string;
        logoUri?: string;
        publishedAt?: string;
        updatedAt?: string;
    }
